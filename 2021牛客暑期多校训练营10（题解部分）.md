### 代码：
```
#include<iostream>
#include<string>
using namespace std;
int main(){
	string ans="10";
	string tem=ans;
	int n;
	cin>>n;
	for(int i=1;i<n;i++){
		for(int j=0;j<tem.size();j++){
			if(tem[j]=='0')tem[j]='1';
			else tem[j]='0';
		}
		ans=ans+tem;
		tem=ans;
	}
	cout<<ans<<endl;
}
```
